InProcessServer#

class ansys.dpf.core.server_types.InProcessServer(ansys_path: str | None = None, as_global: bool = True, load_operators: bool = True, timeout: None = None, context: ansys.dpf.core.server_context.AvailableServerContexts = server_context.SERVER_CONTEXT)#

Server using the InProcess communication protocol.

Import detail#

from ansys.dpf.core.server_types import InProcessServer

Property detail#

property InProcessServer.version#

Get the version of the InProcess server.

Returns:

version – The version of the InProcess server in the format “major.minor”.

Return type:

str

property InProcessServer.os#

Get the operating system of the InProcess server.

Returns:

os – The operating system name. For InProcess servers, it typically returns the current OS, e.g., “posix” or “nt”.

Return type:

str

property InProcessServer.client#

Get the client for the InProcess server.

Returns:

client – InProcess servers do not have a client, so this property returns None.

Return type:

None

property InProcessServer.local_server#

Get whether the InProcess server is running locally.

Returns:

local_server – True, as the InProcess server is always local.

Return type:

bool

property InProcessServer.config#

Get the server configuration for the InProcess server.

Returns:

config – The server configuration for the InProcess server from the AvailableServerConfigs.

Return type:

AvailableServerConfigs
